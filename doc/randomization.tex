\begin{algorithm}[H]
\caption{Operator \texttt{randomizeCandidate}(candidate, perturbationRate)}
\begin{algorithmic}[1]
\For{$i \gets 1$ \textbf{to} \#tasks}
	\If{\texttt{rand()} $<$ perturbationRate}
		\If{task $i$ has no fixed assignment}
			\State assign \texttt{candidate.server\_indices[i]} to a random non-MIST server
			\State \textbf{continue}
		\EndIf
	\EndIf
	\If{\texttt{rand()} $<$ perturbationRate}
		\State \texttt{candidate.priorities[i]} $\gets$ \texttt{candidate.priorities[i]} + $\mathcal{N}(0, 0.05)$
		\State clamp \texttt{candidate.priorities[i]} to the interval $[0,1]$
	\EndIf
\EndFor
\end{algorithmic}
\end{algorithm}
